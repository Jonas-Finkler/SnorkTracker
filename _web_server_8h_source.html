110 bool MyWebServer::begin()
111 {

113  return false;
114  }
115 
116  MyDbg("MyWebServer::begin");
117  WiFi.mode(WIFI_AP_STA);
118  WiFi.softAP("ESP8266AP", "");
119  WiFi.softAPConfig(ip, ip, IPAddress(255, 255, 255, 0));
120  dnsServer.setErrorReplyCode(DNSReplyCode::NoError);
121  dnsServer.start(53, "*", ip);
122  myData->softAPIP = WiFi.softAPIP().toString();
123  myData->softAPmacAddress = WiFi.softAPmacAddress();
124  MyDbg("SoftAPIP address: " + myData->softAPIP, true);
125  MyDbg("SoftAPIP mac address: " + myData->softAPmacAddress, true);
126 
127  WiFi.begin(myOptions->wlanAP.c_str(), myOptions->wlanPassword.c_str());
128  for (int i = 0; i < 20 && WiFi.status() != WL_CONNECTED; i++) { // 10 Sec versuchen
129  MyDbg(".", true, false);
130  myDelay(500);
131  }
132  if (WiFi.status() == WL_CONNECTED) {
133  myData->stationIP = WiFi.localIP().toString();
134  MyDbg("Connected to " + myOptions->wlanAP, true);
135  MyDbg("Station IP address: " + myData->stationIP, true);
136  } else { // switch to AP Mode only
137  MyDbg((String) "No connection to " + myOptions->wlanAP, true);
138  WiFi.disconnect();
139  WiFi.mode(WIFI_AP);
140  }
141

143  server.on("/Main.html", loadMain);
144  server.on("/MainInfo", handleLoadMainInfo);
145  server.on("/FirmwareUpdate.html", loadFirmwareUpdate);
146  server.on("/Settings.html", loadSettings);

290 bool MyWebServer::loadFromSpiffs(String path)
291 {
292  bool ret = false;
293 
294  String dataType = "text/plain";
295  if(path.endsWith("/")) path += "index.htm";
296 
297  if(path.endsWith(".src")) path = path.substring(0, path.lastIndexOf("."));
298  else if(path.endsWith(".html")) dataType = "text/html";
299  else if(path.endsWith(".htm")) dataType = "text/html";
300  else if(path.endsWith(".css")) dataType = "text/css";
301  else if(path.endsWith(".js")) dataType = "application/javascript";
302  else if(path.endsWith(".png")) dataType = "image/png";
303  else if(path.endsWith(".gif")) dataType = "image/gif";
304  else if(path.endsWith(".jpg")) dataType = "image/jpeg";
305  else if(path.endsWith(".ico")) dataType = "image/x-icon";
306  else if(path.endsWith(".xml")) dataType = "text/xml";
307  else if(path.endsWith(".pdf")) dataType = "application/pdf";
308  else if(path.endsWith(".zip")) dataType = "application/zip";
309 
310  File dataFile = SPIFFS.open(path.c_str(), "r");
311  if (dataFile) {
312  if (server.hasArg("download")) {
313  dataType = "application/octet-stream";
314  }
315  if (server.streamFile(dataFile, dataType) == dataFile.size()) {
316  ret = true;
317  }
318  dataFile.close();
319  }
320  return ret;
321 }
