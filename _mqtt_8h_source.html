107 void MyMqtt::reconnect()
108 {
109  MyDbg("Attempting MQTT connection...");
110  // Try 5 times to reconnected
111  for (int i = 0; !PubSubClient::connected() && i < 5; i++) {
112  // Attempt to connect
113  if (PubSubClient::connect(MQTT_NAME, MQTT_USER, MQTT_PASSWORD)) {
114  subscribe(topic_cmd);
115  subscribe(topic_gsm_power);
116  subscribe(topic_gsm_enabled);
117  subscribe(topic_gps_enabled);
118  subscribe(topic_send_on_move_every);
119  subscribe(topic_send_on_non_move_every);
120  MyDbg(" connected");
121  } else {
122  MyDbg(" Failed (" + String(i+1) + ") rc =" + String(state()));
123  MyDbg(" Try again in 5 seconds");
124  // Wait 5 seconds before retrying
125  myDelay(5000);
126  MyDbg(".", false, false);
127  }
128  }
129 }

171 void MyMqtt::handleClient()
172 {
173  if (myGsmGps.isGsmActive) {
174  if (!PubSubClient::connected()) {
175  long currSec = millis() / 1000;
176 
177  if (currSec - mqttLastReconnectSec > myOptions.mqttReconnectIntervalSec) {
178  mqttLastReconnectSec = currSec;
179  reconnect();
180  }
181  }
182  if (connected()) {
183  bool send = false;
184  long currentSec = millis() / 1000;
185 
186  if (myData.isMoving) {
187  send = currentSec - mqttLastSendSec > myOptions.mqttSendOnMoveEverySec;
188  } else {
189  send = currentSec - mqttLastSendSec > myOptions.mqttSendOnNonMoveEverySec;
190  }
191  if (send && sendData()) {
192  mqttLastSendSec = currentSec;
193  }
194  }
195  }
196 }

201 void MyMqtt::mqttCallback(char* topic, byte* payload, unsigned int len)
202 {
203  String strTopic = String((char*)topic);
204 
205  payload[len] = '\0';
206  MyDbg("Message arrived [" + strTopic + "]:[ ");
207  if (len) MyDbg((char *) payload);
208  MyDbg("]");
209 
210  if (MyMqtt::g_myOptions) {
211  if (strTopic == topic_gsm_power) {
212  g_myOptions->gsmPower = atoi((char *) payload);
213  MyDbg(strTopic + g_myOptions->gsmPower ? " - On" : " - Off");
214  }
215  if (strTopic == topic_gsm_enabled) {
216  g_myOptions->isGsmEnabled = atoi((char *) payload);
217  MyDbg(strTopic + g_myOptions->isGsmEnabled ? " - Enabled" : " - Disabled");
218  }
219  if (strTopic == topic_gps_enabled) {
220  g_myOptions->isGpsEnabled = atoi((char *) payload);
221  MyDbg(strTopic + g_myOptions->isGpsEnabled ? " - Enabled" : " - Disabled");
222  }
223  if (strTopic == topic_send_on_move_every) {
224  g_myOptions->mqttSendOnMoveEverySec = atoi((char *) payload);
225  MyDbg(strTopic + " - " + String(g_myOptions->mqttSendOnMoveEverySec));
226  }
227  if (strTopic == topic_send_on_non_move_every) {
228  g_myOptions->mqttSendOnNonMoveEverySec = atoi((char *) payload);
229  MyDbg(strTopic + " - " + String(g_myOptions->mqttSendOnNonMoveEverySec));
230  }
231  }
