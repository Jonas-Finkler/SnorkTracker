175 bool MyDegrees::set(const String &data)
176 {
177  const char *term = data.c_str();
178  uint32_t multiplier = 1000000000UL;
179  uint32_t leftOfDecimal = (uint32_t) atol(term);
180 
181  predecimal = (int16_t)leftOfDecimal;
182  billionths = 0;
183 
184  while (isdigit(*term)) {
185  ++term;
186  }
187 
188  if (*term == '.') {
189  while (isdigit(*++term)) {
190  multiplier /= 10;
191  billionths += (*term - '0') * multiplier;
192  }
193  }
194  negative = false;
195  return true;
196 }

199 double MyLocation::distanceBetween(double lat1, double long1, double lat2, double long2)
200 {
201  // returns distance in meters between two positions, both specified
202  // as signed decimal-degrees latitude and longitude. Uses great-circle
203  // distance computation for hypothetical sphere of radius 6372795 meters.
204  // Because Earth is no exact sphere, rounding errors may be up to 0.5%.
205  // Courtesy of Maarten Lamers
206  double delta = radians(long1 - long2);
207  double sdlong = sin(delta);
208  double cdlong = cos(delta);
209  lat1 = radians(lat1);
210  lat2 = radians(lat2);
211  double slat1 = sin(lat1);
212  double clat1 = cos(lat1);
213  double slat2 = sin(lat2);
214  double clat2 = cos(lat2);
215  delta = (clat1 * slat2) - (slat1 * clat2 * cdlong);
216  delta = sq(delta);
217  delta += sq(clat2 * sdlong);
218  delta = sqrt(delta);
219  double denom = (slat1 * slat2) + (clat1 * clat2 * cdlong);
220  delta = atan2(delta, denom);
221  return delta * 6372795;
222 }

225 double MyLocation::courseTo(double lat1, double long1, double lat2, double long2)
226 {
227  // returns course in degrees (North=0, West=270) from position 1 to position 2,
228  // both specified as signed decimal-degrees latitude and longitude.
229  // Because Earth is no exact sphere, calculated course may be off by a tiny fraction.
230  // Courtesy of Maarten Lamers
231  double dlon = radians(long2 - long1);
232  lat1 = radians(lat1);
233  lat2 = radians(lat2);
234  double a1 = sin(dlon) * cos(lat2);
235  double a2 = sin(lat1) * cos(lat2) * cos(dlon);
236  a2 = cos(lat1) * sin(lat2) - a2;
237  a2 = atan2(a1, a2);
238  if (a2 < 0.0)
239  {
240  a2 += TWO_PI;
241  }
242  return degrees(a2);
243 }
