86 bool MyGsmGps::begin()
87 {
88  if (!myOptions.gsmPower) {
89  MyDbg("sim808 has no power!");
90  return false;
91  }
92 
93  if (!isSimActive) {
94  MyDbg("MyGsmGps::begin");
95  myData.status = "Sim808 Initializing...";
96  MyDbg(myData.status);
97  for (int i = 0; !gsmSim808.restart() && i <= 5; i++) {
98  if (!myOptions.gsmPower) {
99  MyDbg("Sim808 Initializing ... canceled");
100  return false;
101  }
102  if (i == 5) { // not working!
103  myData.status = "Sim808 restart failed";
104  MyDbg(myData.status);
105  return false;
106  }
107  MyDbg(".", false, false);
108  myDelay(500);
109  }
110  myData.status = "Sim808 connected";
111  MyDbg(myData.status);
112 
113  gsmSim808.setBaud(9600);
114  isSimActive = true;
115  }
116 
117  if (isSimActive && myOptions.isGpsEnabled && !isGpsActive) {
118  enableGps(true);
119  isGpsActive = true;
120  }
121 
122  if (isSimActive && myOptions.isGsmEnabled && !isGsmActive) {
123  myData.status = "Sim808 Waiting for network...";
124  MyDbg(myData.status);
125  for (int i = 0; !gsmSim808.waitForNetwork() && i <= 5; i++) {
126  if (!myOptions.gsmPower) {
127  MyDbg("Sim808 Waiting for network... canceled");
128  return false;
129  }
130  if (i == 5) { // not working!
131  myData.status = "Sim808 network failed";
132  MyDbg(myData.status);
133  return false;
134  }
135  MyDbg(".", false, false);
136  myDelay(500);
137  }
138  if (!gsmSim808.isNetworkConnected()) {
139  myData.status = "Sim808 network failed";
140  MyDbg(myData.status);
141  }
142 
143  MyDbg((String) "GPRS: " + myOptions.gprsAP);
144  if (!gsmSim808.gprsConnect(myOptions.gprsAP.c_str(), "", "")) {
145  myData.status = "Sim808 gprs connection failed!";
146  MyDbg(myData.status);
147  while (true);
148  }
149  myData.status = "Sim808 gsm connected";
150  MyDbg(myData.status);
151 
152  myData.modemInfo = gsmSim808.getModemInfo();
153  MyDbg("Modem info: " + myData.modemInfo);
154 
155  myData.modemIP = gsmSim808.getLocalIP();
156  MyDbg("Modem IP: " + myData.modemIP);
157 
158  myData.imei = gsmSim808.getIMEI();
159  MyDbg("sim808: " + myData.modemInfo);
160 
161  myData.cop = gsmSim808.getOperator();
162  MyDbg("cop: " + myData.modemInfo);
163 
164  isGsmActive = true;
165  }
166  return true;
167 }

170 void MyGsmGps::handleClient()
